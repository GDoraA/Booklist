/**

 *

 * - kideríteni, hogy a hibás "A Jefferson-k�d" cím már a letöltött HTML-ben szerepel-e így
 * - vagy csak az extractFirstMatch_() / parser feldolgozás közben sérül
 *
 * Mit vizsgál:
 * - HTML hossz
 * - tartalmaz-e "�" replacement karaktert
 * - van-e charset deklaráció a HTML-ben
 * - mit tartalmaz nyersen az og:title meta mező
 * - mit ad vissza az extractFirstMatch_() az og:title mezőből
 * - mit tartalmaz nyersen a h1 tag
 * - mit ad vissza az extractFirstMatch_() a h1 tagből
 * - jelenleg melyik címforrást választaná a parser
 *
 * Nem módosít adatot, csak Logger.log() kimenetet ad.
 */
function testLibriProductTitleSourceDebugJeffersonKod() {
  // Tesztelt Libri termékoldal
  const url = "https://www.libri.hu/konyv/steve_berry.a-jefferson-kod--2.html";

  // A jelenlegi éles HTML-letöltő helperrel kérjük le az oldalt,
  // hogy pontosan ugyanazt lássuk, amit a parser is kap.
  const html = fetchPublisherHtml_(url);

  // Biztonságosan stringgé alakítjuk a választ.
  const text = String(html || "");
  const lower = text.toLowerCase();

  Logger.log("PRODUCT URL:");
  Logger.log(url);

  Logger.log("HTML LENGTH:");
  Logger.log(text.length);

  // Ha ez true, akkor a sérült karakter már a letöltött HTML szövegben jelen van.
  Logger.log("CONTAINS REPLACEMENT CHAR:");
  Logger.log(text.indexOf("�") !== -1);

  // Charset deklaráció keresése a HTML-ben.
  // Ez segíthet eldönteni, milyen karakterkódolást hirdet az oldal.
  const charsetMatch = text.match(/charset=["']?([^"'\s>]+)/i);

  Logger.log("CHARSET MATCH:");
  Logger.log(charsetMatch ? charsetMatch[0] : "NINCS CHARSET TALÁLAT");

  // Ugyanaz az og:title regex, amit a Libri parser is használ címkinyerésre.
  const ogTitleRegex = /<meta property="og:title" content="([^"]+)"/i;

  // Nyers regex találat: itt még nincs stripHtml_() / decodeHtmlEntities_().
  const ogTitleRawMatch = text.match(ogTitleRegex);

  Logger.log("OG TITLE RAW:");
  Logger.log(ogTitleRawMatch ? ogTitleRawMatch[1] : "NINCS OG TITLE TALÁLAT");

  // extractFirstMatch_() eredménye az og:title mezőre.
  const ogTitleExtracted = extractFirstMatch_(text, ogTitleRegex);

  Logger.log("OG TITLE EXTRACTED:");
  Logger.log(ogTitleExtracted || "NINCS OG TITLE EXTRACTED");

  // Az og:title körüli HTML-részlet kiírása,
  // hogy látható legyen a tényleges forráskörnyezet.
  const ogTitlePos = lower.indexOf('property="og:title"');

  Logger.log("OG TITLE POSITION:");
  Logger.log(ogTitlePos);

  if (ogTitlePos >= 0) {
    Logger.log("HTML AROUND OG TITLE:");
    Logger.log(text.substring(Math.max(0, ogTitlePos - 300), ogTitlePos + 700));
  }

  // h1 regex: fallback címforrás, ha nincs og:title találat.
  const h1Regex = /<h1[^>]*>([\s\S]*?)<\/h1>/i;

  // Nyers h1 találat.
  const h1RawMatch = text.match(h1Regex);

  Logger.log("H1 RAW:");
  Logger.log(h1RawMatch ? h1RawMatch[1] : "NINCS H1 TALÁLAT");

  // extractFirstMatch_() eredménye h1-re.
  const h1Extracted = extractFirstMatch_(text, h1Regex);

  Logger.log("H1 EXTRACTED:");
  Logger.log(h1Extracted || "NINCS H1 EXTRACTED");

  // A h1 körüli HTML-részlet kiírása.
  const h1Pos = lower.indexOf("<h1");

  Logger.log("H1 POSITION:");
  Logger.log(h1Pos);

  if (h1Pos >= 0) {
    Logger.log("HTML AROUND H1:");
    Logger.log(text.substring(Math.max(0, h1Pos - 300), h1Pos + 700));
  }

  // Ez modellezi a jelenlegi parser címválasztását:
  // először og:title, ha nincs, akkor h1.
  const parserTitle =
    extractFirstMatch_(text, ogTitleRegex) ||
    extractFirstMatch_(text, h1Regex);

  Logger.log("CURRENT PARSER TITLE RESULT:");
  Logger.log(parserTitle || "NINCS PARSER TITLE");
}
/**
 * Libri termékoldal HTML debug a datas blokk és a termékadat mezők környezetére.
 *
 * Cél:
 * - megkeresni, hogy a Libri HTML-ben pontosan hol szerepel a datas blokk
 * - ellenőrizni, hogy a fontos mezők milyen néven és milyen szerkezetben vannak jelen
 * - előkészíteni az extractLibriDatasField_() helper pontosítását
 *
 * Vizsgált mezők:
 * - datas
 * - kiadó
 * - Fordító
 * - kiadás éve
 * - oldalak száma
 * - isbn
 * - nyelv
 *
 * Nem módosít adatot, csak Logger.log() kimenetet ad.
 */
function testLibriPublisherHtmlDebugJeffersonKod() {
  const url = "https://www.libri.hu/konyv/steve_berry.a-jefferson-kod--2.html";

  // A Libri-specifikus, helyes karakterkódolású letöltőt használjuk,
  // hogy ugyanazt a HTML-t lássuk, amit a parser is kap.
  const html = fetchLibriHtml_(url);
  const text = String(html || "");
  const lower = text.toLowerCase();

  Logger.log("PRODUCT URL:");
  Logger.log(url);

  Logger.log("HTML LENGTH:");
  Logger.log(text.length);

  /**
   * Egy adott keresett szöveg első előfordulása körüli HTML-részletet logolja.
   *
   * Paraméterek:
   * - label: emberileg olvasható log címke
   * - needle: keresett szöveg
   * - before: hány karaktert írjon ki az előfordulás előtt
   * - after: hány karaktert írjon ki az előfordulás után
   */
  function logAround(label, needle, before, after) {
    const needleLower = String(needle || "").toLowerCase();
    const pos = lower.indexOf(needleLower);

    Logger.log("----- " + label + " -----");
    Logger.log(label + " POSITION:");
    Logger.log(pos);

    if (pos >= 0) {
      Logger.log("HTML AROUND " + label + ":");
      Logger.log(
        text.substring(
          Math.max(0, pos - (before || 1000)),
          Math.min(text.length, pos + (after || 2500))
        )
      );
    } else {
      Logger.log("NINCS TALÁLAT ERRE: " + needle);
    }
  }

  /**
   * Egy mező összes előfordulásából kilogolja az első néhány pozíciót
   * és azok környezetét.
   *
   * Ez azért kell, mert például a "kiadó" szó szerepelhet meta keywordsben,
   * datas blokkban vagy más HTML-részben is.
   */
  function logAllOccurrences(label, needle, maxCount) {
    const needleLower = String(needle || "").toLowerCase();
    let fromIndex = 0;
    let count = 0;

    Logger.log("===== ALL OCCURRENCES: " + label + " / " + needle + " =====");

    while (count < (maxCount || 5)) {
      const pos = lower.indexOf(needleLower, fromIndex);

      if (pos === -1) {
        if (count === 0) {
          Logger.log("NINCS ELŐFORDULÁS.");
        }
        break;
      }

      Logger.log(label + " OCCURRENCE #" + (count + 1) + " POSITION:");
      Logger.log(pos);

      Logger.log("HTML AROUND " + label + " OCCURRENCE #" + (count + 1) + ":");
      Logger.log(
        text.substring(
          Math.max(0, pos - 700),
          Math.min(text.length, pos + 1400)
        )
      );

      fromIndex = pos + needleLower.length;
      count++;
    }
  }

  // 1) Datas blokk környezetének keresése.
  logAround("DATAS", "datas", 1500, 5000);

  // 2) Fontos datas mezőnevek első előfordulásának környezete.
  logAround("KIADO_LOWER", "kiadó", 1000, 2500);
  logAround("FORDITO_CAPITAL", "Fordító", 1000, 2500);
  logAround("KIADAS_EVE", "kiadás éve", 1000, 2500);
  logAround("OLDALAK_SZAMA", "oldalak száma", 1000, 2500);
  logAround("ISBN", "isbn", 1000, 2500);
  logAround("NYELV", "nyelv", 1000, 2500);

  // 3) Az első néhány előfordulás külön is, hogy lássuk,
  // melyik találat van meta részben és melyik tényleges termékadatban.
  logAllOccurrences("KIADO_ALL", "kiadó", 5);
  logAllOccurrences("FORDITO_ALL", "Fordító", 5);
  logAllOccurrences("KIADAS_EVE_ALL", "kiadás éve", 5);
  logAllOccurrences("OLDALAK_SZAMA_ALL", "oldalak száma", 5);
  logAllOccurrences("ISBN_ALL", "isbn", 5);
  logAllOccurrences("NYELV_ALL", "nyelv", 5);

  // 4) Gyors ellenőrzés: a jelenlegi datas helper mit talál ezekre.
  Logger.log("===== CURRENT extractLibriDatasField_ RESULTS =====");

  Logger.log("kiadó:");
  Logger.log(extractLibriDatasField_(text, "kiadó"));

  Logger.log("Fordító:");
  Logger.log(extractLibriDatasField_(text, "Fordító"));

  Logger.log("kiadás éve:");
  Logger.log(extractLibriDatasField_(text, "kiadás éve"));

  Logger.log("oldalak száma:");
  Logger.log(extractLibriDatasField_(text, "oldalak száma"));

  Logger.log("isbn:");
  Logger.log(extractLibriDatasField_(text, "isbn"));

  Logger.log("nyelv:");
  Logger.log(extractLibriDatasField_(text, "nyelv"));
}
